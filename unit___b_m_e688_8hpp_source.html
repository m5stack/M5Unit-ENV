168 {
169 value = (value & ~0x3F) | (s & 0x3F);

171 inline void factor(const Factor f)
172 {
173 value = (value & ~(0x03 << 6)) | (m5::stl::to_underlying(f) << 6);

179 static uint8_t from(const uint16_t duration)
180 {
181 uint8_t f{};
182 uint16_t d{duration};
183 while (d > 0x3F) {
184 d >>= 2;
185 ++f;
186 }
187 return (f <= 0x03) ? ((uint8_t)d | (f << 6)) : 0xFF;
188 }
191 static uint16_t to(const uint8_t v)
192 {
193 constexpr uint16_t tbl[] = {1, 4, 16, 64};
194 return (v & 0x3F) * tbl[(v >> 6) & 0x03];
195 }
